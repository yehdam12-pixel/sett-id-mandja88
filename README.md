    <script>
        // Data game populer
        const popularGames = [
            { name: "Sweet Bonanza", icon: "fas fa-candy-cane" },
            { name: "Starlight Princess", icon: "fas fa-crown" },
            { name: "Gates of Olympus", icon: "fas fa-monument" },
            { name: "Mahjong Ways", icon: "fas fa-chess-board" },
            { name: "Aztec Gems", icon: "fas fa-gem" },
            { name: "Great Rhino", icon: "fas fa-hippo" },
            { name: "Wolf Gold", icon: "fas fa-paw" },
            { name: "Buffalo King", icon: "fas fa-bullhorn" }
        ];
        
        // Inisialisasi daftar game populer
        const gameListContainer = document.getElementById('gameList');
        const gameSelect = document.getElementById('gameSelect');
        
        popularGames.forEach(game => {
            const gameItem = document.createElement('div');
            gameItem.className = 'game-item';
            gameItem.innerHTML = `
                <div class="game-icon">
                    <i class="${game.icon}"></i>
                </div>
                <div class="game-name">${game.name}</div>
            `;
            
            gameItem.addEventListener('click', () => {
                // Hapus kelas selected dari semua item
                document.querySelectorAll('.game-item').forEach(item => {
                    item.classList.remove('selected');
                });
                
                // Tambahkan kelas selected ke item yang diklik
                gameItem.classList.add('selected');
                
                // Set nilai dropdown
                gameSelect.value = game.name;
            });
            
            gameListContainer.appendChild(gameItem);
        });
        
        // Tangani pengiriman formulir
        const form = document.getElementById('idSettingForm');
        const resultBox = document.getElementById('resultBox');
        const initialMessage = document.getElementById('initialMessage');
        
        form.addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Ambil nilai dari formulir
            const username = document.getElementById('username').value;
            const slotId = document.getElementById('slotId').value;
            const game = document.getElementById('gameSelect').value;
            
            // Validasi
            if (!username || !slotId || !game) {
                alert('Harap isi semua field yang diperlukan!');
                return;
            }
            
            // Tampilkan hasil
            document.getElementById('resultUsername').textContent = username;
            document.getElementById('resultSlotId').textContent = slotId;
            document.getElementById('resultGame').textContent = game;
            
            // Tampilkan kotak hasil dan sembunyikan pesan awal
            initialMessage.style.display = 'none';
            resultBox.style.display = 'block';
            
            // Scroll ke hasil
            resultBox.scrollIntoView({ behavior: 'smooth' });
            
            // Reset seleksi game
            document.querySelectorAll('.game-item').forEach(item => {
                item.classList.remove('selected');
            });
            
            // Reset formulir (opsional)
            // form.reset();
        });
        
        // Sinkronisasi dropdown dengan seleksi game populer
        gameSelect.addEventListener('change', function() {
            const selectedGame = this.value;
            
            // Hapus seleksi dari semua item game
            document.querySelectorAll('.game-item').forEach(item => {
                item.classList.remove('selected');
                
                // Jika nama game di item cocok dengan nilai dropdown, tandai sebagai terpilih
                if (item.querySelector('.game-name').textContent === selectedGame) {
                    item.classList.add('selected');
                }
            });
        });
        
        // Animasi tambahan untuk pengalaman yang lebih baik
        document.querySelectorAll('.game-item').forEach(item => {
            item.addEventListener('mouseenter', function() {
                this.style.transform = 'translateY(-5px)';
            });
            
            item.addEventListener('mouseleave', function() {
                if (!this.classList.contains('selected')) {
                    this.style.transform = 'translateY(0)';
                }
            });
        });
    </script>
